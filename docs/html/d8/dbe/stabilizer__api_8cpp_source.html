45 std::string SMTStabilizer::run_pipeline(stabilizer::parser::Parser &parser, const SMTStabilizerOptions &options) {
46  configure_parser(parser, options);
47 
48  stabilizer::node::NodeManager nm(parser);
49  nm.simplify_assertions();
50 
51  stabilizer::kernel::Kernel kernel(nm, options.get_context_propagation(), options.get_subgraph_pruning());
52  kernel.apply(nm);
53 
54  return nm.to_string();
55 }
