47 Kernel::Kernel(node::NodeManager &nm, const bool &context_propagation, const bool &symmetry_breaking_perturbation) : d_context_propagation(context_propagation), d_symmetry_breaking_perturbation(symmetry_breaking_perturbation) {
48  std::vector<node::Node> visit(nm.assertions());
49  std::unordered_map<node::Node, size_t> node2index;
50  // node2index.emplace(nm.assertion(), 0);
51 
52  while (!visit.empty()) {
53  auto node = visit.back();
54  auto [itr, success] = node2index.emplace(node, SIZE_MAX);
55  if (success) {
56  for (const auto &child : node->getChildren()) {
57  visit.emplace_back(child);
58  }
59  if (node->isUFApplication())
60  visit.emplace_back(nm.mkUFVNode(node->getName(), node->getSort()));
61  }
62  else if (itr->second == SIZE_MAX) {
63  // std::cout << node->toString() << ' ' << node->getSort()->isDec() << std::endl;
64  visit.pop_back();
65  itr->second = d_nodes.size();
66  d_nodes.emplace_back(node);
67  d_is_commutative.emplace_back(is_commutative(d_nodes.size() - 1, false));
68  d_is_symbol.emplace_back(node->isVar() || (node->isArray() && !node->isConstArray()) || node->isVUF() || node->isUFName());
69  d_graph.emplace_back();
70  d_reversed_graph.emplace_back();
71  if (d_is_symbol.back())
72  d_symbols.emplace_back(d_nodes.size() - 1);
73 
74  d_hash_table.emplace_back(util::hash_node(node));
75  for (size_t i = 0, isz = node->getChildrenSize(); i < isz; ++i) {
76  const auto &child = node->getChild(i);
77  const size_t &child_index = node2index.at(child);
78  d_graph.back().emplace_back(child_index);
79  if (child->getKind() == stabilizer::parser::NODE_KIND::NT_FP_ADD ||
80  child->getKind() == stabilizer::parser::NODE_KIND::NT_FP_MUL ||
81  child->getKind() == stabilizer::parser::NODE_KIND::NT_EXISTS ||
82  child->getKind() == stabilizer::parser::NODE_KIND::NT_FORALL)
83  d_reversed_graph.at(child_index).emplace_back(d_nodes.size() - 1, std::min(i, 0ul));

86  }
87  if (node->isUFApplication()) {
88  const size_t &uf_index = node2index.at(nm.mkUFVNode(node->getName(), node->getSort()));
89  d_graph.back().emplace_back(uf_index);
90  d_reversed_graph.at(uf_index).emplace_back(d_nodes.size() - 1, node->getChildrenSize());
91  }

95  }

99  d_visited.resize(d_nodes.size(), false);
100  d_symbol_num = d_symbols.size();
101 
102  for (const auto &assertion : nm.assertions()) {
103  util::hash_combine(d_hash_table.at(node2index.at(assertion)), static_cast<size_t>(parser::NODE_KIND::NT_AND));

109 bool Kernel::is_commutative(const size_t &i, const bool &from_cache) {
110  // return false;
111  if (from_cache)
112  return d_is_commutative.at(i);
113  const auto &node = d_nodes.at(i);
114  switch (node->getKind()) {
115  case stabilizer::parser::NODE_KIND::NT_ADD:
116  case stabilizer::parser::NODE_KIND::NT_MUL:
117  case stabilizer::parser::NODE_KIND::NT_AND:
118  case stabilizer::parser::NODE_KIND::NT_IAND:
119  case stabilizer::parser::NODE_KIND::NT_OR:
120  case stabilizer::parser::NODE_KIND::NT_XOR:
121  case stabilizer::parser::NODE_KIND::NT_BV_ADD:
122  case stabilizer::parser::NODE_KIND::NT_BV_MUL:
123  case stabilizer::parser::NODE_KIND::NT_BV_AND:
124  case stabilizer::parser::NODE_KIND::NT_BV_OR:
125  case stabilizer::parser::NODE_KIND::NT_BV_XOR:
126  case stabilizer::parser::NODE_KIND::NT_EQ:
127  case stabilizer::parser::NODE_KIND::NT_DISTINCT:
128  case stabilizer::parser::NODE_KIND::NT_MAX:
129  case stabilizer::parser::NODE_KIND::NT_MIN:
130  case stabilizer::parser::NODE_KIND::NT_BV_NAND:
131  case stabilizer::parser::NODE_KIND::NT_BV_NOR:
132  case stabilizer::parser::NODE_KIND::NT_BV_COMP:
133  case stabilizer::parser::NODE_KIND::NT_BV_SADDO:
134  case stabilizer::parser::NODE_KIND::NT_BV_UADDO:
135  case stabilizer::parser::NODE_KIND::NT_BV_UMULO:
136  case stabilizer::parser::NODE_KIND::NT_BV_XNOR:
137  case stabilizer::parser::NODE_KIND::NT_FP_EQ:
138  case stabilizer::parser::NODE_KIND::NT_FP_MIN:
139  case stabilizer::parser::NODE_KIND::NT_FP_MAX:
140 
141  // Quantifiers are treated as commutative for hashing purposes
142  case stabilizer::parser::NODE_KIND::NT_FORALL:
143  case stabilizer::parser::NODE_KIND::NT_EXISTS:
144 
145  // FP_ADD, FP_MUL, SUB, DIV_REAL are commutative from index one
146  case stabilizer::parser::NODE_KIND::NT_FP_ADD:
147  case stabilizer::parser::NODE_KIND::NT_FP_MUL:
148  // case stabilizer::parser::NODE_KIND::NT_SUB:
149  // case stabilizer::parser::NODE_KIND::NT_DIV_REAL:
150  return true;
151  default:
152  return false;
153  }

156 void Kernel::propagate(const std::vector<size_t> &processing) {
157  if (!d_context_propagation) {
158  static std::vector<size_t> pre_hash_table = d_hash_table;
159  for (const auto &node : processing)
160  pre_hash_table.at(node) = d_hash_table.at(node);
161  std::vector<size_t> v(processing);
162  for (const auto &node : v) {
163  const auto &children = d_graph.at(node);
164  if (children.empty()) continue;
165  if (is_commutative(node)) {
166  util::hash_communative_children(pre_hash_table.at(node), children, d_hash_table);
167  }
168  else {
169  util::hash_children(pre_hash_table.at(node), children, d_hash_table);
170  }
171  util::hash_parents(pre_hash_table.at(node), d_reversed_graph.at(node), d_hash_table);
172  }
173 
174  for (const auto &node : v)
175  d_hash_table.at(node) = pre_hash_table.at(node);
176  }
177  else {
178  for (const auto &node : processing) {
179  const auto &children = d_graph.at(node);
180  if (children.empty()) continue;
181  if (is_commutative(node)) {
182  util::hash_communative_children(d_hash_table.at(node), children, d_hash_table);
183  }
184  else {
185  util::hash_children(d_hash_table.at(node), children, d_hash_table);
186  }
187  }
188  for (auto it = processing.rbegin(); it != processing.rend(); ++it) {
189  const auto &node = *it;
190  const auto &parents = d_reversed_graph.at(node);
191  if (parents.empty())
192  continue;
193  util::hash_parents(d_hash_table.at(node), parents, d_hash_table);
194  }
195  }
196 }

204 void Kernel::context_propagate(const std::vector<size_t> &processing, const std::vector<size_t> &symbols) {
205  size_t hash_count = 0, prev_hash_count = 0;
206  std::vector<size_t> unique_hashes(symbols.size());
207  do {
208  prev_hash_count = hash_count;
209  propagate(processing);
210  std::transform(symbols.begin(), symbols.end(), unique_hashes.begin(), [this](size_t index) { return d_hash_table.at(index); });
211  std::sort(unique_hashes.begin(), unique_hashes.end());
212  hash_count = 1;
213  for (size_t i = 1, isz = unique_hashes.size(); i < isz; ++i)
214  if (unique_hashes[i] != unique_hashes[i - 1])
215  ++hash_count;
216  } while (prev_hash_count != hash_count);
217 }

228 void Kernel::specific_propagate() {
229  std::unordered_map<size_t, size_t> hash_count;
230 
231  for (const auto &node : d_symbols) {
232  if (d_visited.at(node))
233  continue;
234  std::vector<size_t> processing;
235  std::vector<size_t> symbols;
236  d_visited.at(node) = true;
237  std::vector<size_t> stack{node};
238  while (!stack.empty()) {
239  auto current = stack.back();
240  stack.pop_back();
241  processing.emplace_back(current);
242  if (d_is_symbol.at(current)) {
243  symbols.emplace_back(current);
244  // std::cout << d_hash_table.at(current) << ' ' << d_nodes.at(current)->toString() << std::endl;
245  }
246  for (const auto &[father_index, oper_index] : d_reversed_graph.at(current)) {
247  if (!d_visited.at(father_index)) {
248  d_visited.at(father_index) = true;
249  stack.emplace_back(father_index);
250  }
251  }
252  for (const auto &child_index : d_graph.at(current)) {
253  if (!d_visited.at(child_index)) {
254  d_visited.at(child_index) = true;
255  stack.emplace_back(child_index);
256  }
257  }
258  }
259 
260  size_t symbols_hash = 0;
261  util::hash_communative_children(symbols_hash, symbols, d_hash_table);
262  hash_count[symbols_hash]++;
263  size_t count = hash_count[symbols_hash];
264  for (const auto &node : processing) {
265  util::hash_combine(d_hash_table.at(node), symbols_hash);
266  util::hash_combine(d_hash_table.at(node), count);
267  }
268 
269  std::vector<size_t> symbol_hashes(symbols.size());
270  std::transform(symbols.begin(), symbols.end(), symbol_hashes.begin(), [this](size_t index) { return d_hash_table.at(index); });
271  std::sort(symbol_hashes.begin(), symbol_hashes.end());
272  size_t idx_hash = 0;
273  bool is_select = false;
274  for (size_t i = 1, isz = symbol_hashes.size(); i < isz; ++i) {
275  if (symbol_hashes[i] == symbol_hashes[i - 1]) {
276  idx_hash = symbol_hashes.at(i);
277  is_select = true;
278  break;
279  }
280  }
281 
282  if (is_select) {
283  size_t idx = 0;
284  for (size_t i = 0, isz = symbols.size(); i < isz; ++i) {
285  if (d_hash_table.at(symbols.at(i)) == idx_hash) {
286  idx = symbols.at(i);
287  break;
288  }
289  }
290  util::hash_combine(d_hash_table.at(idx), 0);
291  std::sort(processing.begin(), processing.end());
292  context_propagate(processing, symbols);
293  }
294  }
295  for (const auto &node : d_processing)
296  d_visited.at(node) = false;
297 }

305 void Kernel::rebuild_graph() {
306  std::unordered_map<size_t, size_t> hash_count;
307  for (const auto &node : d_processing)
308  hash_count[d_hash_table.at(node)]++;
309  for (const auto &node : d_processing) {
310  if (hash_count.at(d_hash_table.at(node)) > 1)
311  d_visited.at(node) = true;
312  }
313 
314  d_symbols.clear();
315  std::vector<size_t> unique_symbols;
316  std::vector<size_t> new_processing;
317  for (const auto &node : d_processing) {
318  if (!d_visited.at(node)) {
319  d_graph.at(node).clear();
320  d_reversed_graph.at(node).clear();
321  if (d_is_symbol.at(node))
322  unique_symbols.emplace_back(node);
323  continue;
324  }
325  new_processing.emplace_back(node);
326  d_graph.at(node).erase(std::remove_if(d_graph.at(node).begin(), d_graph.at(node).end(), [this](const size_t &child_index) {
327  return !d_visited.at(child_index);
328  }),
329  d_graph.at(node).end());
330  d_reversed_graph.at(node).erase(std::remove_if(d_reversed_graph.at(node).begin(), d_reversed_graph.at(node).end(), [this](const std::pair<size_t, size_t> &father_info) {
331  return !d_visited.at(father_info.first);
332  }),
333  d_reversed_graph.at(node).end());
334  if (d_is_symbol.at(node) && d_visited.at(node))
335  d_symbols.emplace_back(node);
336  }
337  std::sort(unique_symbols.begin(), unique_symbols.end(), [this](const size_t &a, const size_t &b) {
338  if (d_context_hash.at(a) == d_context_hash.at(b))
339  return d_hash_table.at(a) < d_hash_table.at(b);
340  return d_context_hash.at(a) < d_context_hash.at(b);
341  });
342  for (const auto &node : unique_symbols) {
343  d_unique_symbols.emplace_back(node);
344  }
345 
346  for (const auto &node : d_processing)
347  d_visited.at(node) = false;
348  d_processing.swap(new_processing);
349  // std::cout << d_processing.size() << std::endl;
350 }

359 void Kernel::sort_propagate(std::unordered_map<std::string, node::Sort> &sort_key_map, std::vector<std::vector<parser::Parser::DTTypeDecl>> &datatype_blocks) {

361  std::unordered_map<node::Sort, size_t> tmp_dts;
362  size_t _cnt = 0;
363  for (auto &block : datatype_blocks) {
364  ++_cnt;
365  for (size_t i = 0, isz = block.size(); i < isz; ++i) {
366  auto &td = block.at(i);

379  if (d_nodes.at(i)->getSort()->isDec()) {
380  if (sort_idx.find(d_nodes.at(i)->getSort()) == sort_idx.end()) {
381  sort_idx[d_nodes.at(i)->getSort()] = sort_idx.size();

393  }
